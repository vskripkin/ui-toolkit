Checkfield

С легкостью создавайте radio и checkbox


Наши radio/checkbox принимают 5 зарезервированых параметров:


  • @type - тип элемента, принимает значение radio или checkbox
  • @font-size - размер шрифта
  • @pseudo - class для псеудо, стандартное значение "pseudo"
  • @label - class для лейбла, стандартное значение "label"
  • @prefix - class для преффикса, стандартное значение "__"
.my-checkfield { #ui.checkfield ( @type, @font-size, @pseudo, @label, @prefix ); }

less code

.checkbox
{
	#ui.checkfield(checkbox);

	.states() // описываем состояния
	{
		.default() // обычное состояние
		{
			color: @color-primary;
			border-radius: 0.2rem;
			box-shadow: inset 0 0 0 0.1em currentColor;
			transition: color 0.3s cubic-bezier(0.23,1,0.32,1),
						box-shadow 0.3s cubic-bezier(0.23,1,0.32,1);
		}
		.hover() // состояние при наведении курсора
		{

		}
		.active() // активное состояние
		{

		}
		.checked() // состояние checked
		{
			box-shadow: inset 0 0 0 0.6em currentColor;
		}
		.disabled() // состояние disabled
		{
			color: #666;
			background-color: currentColor;
			opacity: 0.4;
		}
		.disabled-checked() // состояние disabled checked
		{
			opacity: 1;
		}
	}
}

less code

.radio
{
	#ui.checkfield(radio);

	.states() // описываем состояния
	{
		.default() // обычное состояние
		{
			border-radius: 50%;
			box-shadow: inset 0 0 0 0.1em currentColor;
			transition: color 0.3s cubic-bezier(0.23,1,0.32,1);
			will-change: color;
			color: @color-primary;

			&:after
			{
				width: 100%;
				height: 100%;
				transition: transform 0.3s cubic-bezier(0.23,1,0.32,1),
							opacity   0.3s cubic-bezier(0.23,1,0.32,1);
				will-change: transform, opacity;
				transform: scale(0);
				opacity: 1;
			}
		}
		.hover() // состояние при наведении курсора
		{
			&:after
			{
				opacity: 0.5;
				transform: scale(0.45);
			}
		}
		.checked() // состояние checked
		{
			&:after
			{
				opacity: 1 !important;
				transform: scale(0.5) !important;
			}
		}
		.disabled() // состояние disabled
		{
			opacity: 0.4;

			&:after
			{
				opacity: 0;
			}
		}
		.disabled-checked() // состояние disabled checked
		{
			opacity: 0.4;
		}
	}
}
